<script>
// --- Confetti and Animation Logic ---
const canvas = document.getElementById("canvas");
const ctx = canvas.getContext("2d");
let width, height, particles = [];
// Ensure canvas ts screen
function resize() {
width = canvas.width = window.innerWidth;
height = canvas.height = window.innerHeight;
}
window.addEventListener('resize', resize);
resize();
// Particle denition
class Particle {
constructor() {
this.x = Math.random() * width;
this.y = Math.random() * height - height;
this.size = Math.random() * 8 + 4;
this.speedY = Math.random() * 3 + 1;
this.speedX = Math.random() * 2 - 1;
// Random vibrant colors
this.color = `hsl(${Math.random() * 360}, 80%, 60%)`;
}
update() {
this.y += this.speedY;
this.x += this.speedX;
// Reset if falls off screen
if (this.y > height) {
this.y = -10;
this.x = Math.random() * width;
}
}
draw() {
ctx.llStyle = this.color;
ctx.beginPath();
ctx.arc(this.x, this.y, this.size, 0, Math.PI * 2);
ctx.ll();
}
}
// Create particles
function initParticles() {
for (let i = 0; i < 150; i++) {
particles.push(new Particle());
}
}
// Animate loop
function animateParticles() {
ctx.clearRect(0, 0, width, height);
particles.forEach(p => {
p.update();
p.draw();
});
requestAnimationFrame(animateParticles);
}
// --- Button Click Action ---
function startCelebration() {
const btn = document.getElementById('openBtn');
const card = document.getElementById('card');
// Fade out button
btn.style.opacity = '0';
setTimeout(() => {
// Hide button completely
btn.classList.add('hidden');
// Show card
card.style.display = 'block';
// Start Confetti
initParticles();
animateParticles();
// Change background to celebratory colors
document.body.style.background = "linear-gradient(135deg, #4A00E0, #8E2DE2)";
document.body.style.transition = "background 2s ease";
}, 500);
}
